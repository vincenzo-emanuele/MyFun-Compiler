fun stampa(string messaggio)
    ?. messaggio;
end fun;

fun leggiOp(): string
    string op;
    % op;
    return op;
end fun;

fun leggiReal(): real
    real num;
    % num;
    return num;
end fun;

fun leggiInt(): integer
    integer num;
    % num;
    return num;
end fun;

fun doOp(real first, real second, string op): real
    if op = '+' then
        return first + second;
    end if;

    if op = '-' then
        return first - second;
    end if;

    if op = '/' and second != 0 then
        return first / second;
    else
        return 0;
    end if;

    if op = '*' then
        return first * second;
    end if;
end fun;

main
    string op;
    integer ask;
    real first, second;
    ask <- 1;
    while ask = 1 do
        stampa('scegli l operazione che vuoi effettuare (+, -, *, /)'); //chiedo all'utente di inserire un numero
        op <- leggiOp();
        stampa('Inserisci primo numero');
        first <- leggiReal();
        stampa('Inserisci secondo numero');
        second <- leggiReal();
        stampa(doOp(first, second, op));
        stampa('Vuoi continuare (si=1, no=0)');
        ask <- leggiInt();
    end do;
end main;
